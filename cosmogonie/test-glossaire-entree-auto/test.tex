	 \newcommand{\newentry}[2]
	 {
	 	\newglossaryentry{#1}{
	 		name = {#1},
	 		description={#2}
	 	}
	 }
	 
	 % #1: Nom commande pour appeler dans le texte (ex: Dieu) #2 : Nom de l'entité qui apparaitra dans le texte (et sera également en entrée du Glossaire) #3: Description dans le glossaire
	 %

	{

	}
